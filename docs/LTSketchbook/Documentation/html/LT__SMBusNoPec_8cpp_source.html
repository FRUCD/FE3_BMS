92 void LT_SMBusNoPec::writeWord(uint8_t address, uint8_t command, uint16_t data)
93 {
94  uint8_t buffer[2];
95  buffer[0] = (uint8_t) (data & 0xff);
96  buffer[1] = (uint8_t) (data >> 8);
97 
98 #if USE_BLOCK_TRANSACTION
99  if (i2cbus.writeBlockData(address, command, 2, buffer))
100  Serial.print(F("Write Word: fail.\n"));
101 #else
102  uint16_t rdata;
103  rdata = (data << 8) | (data >> 8);
104  if (i2cbus.writeWordData(address, command, rdata))
105  Serial.print(F("Write Word: fail.\n"));

110 {

120  uint16_t rdata;
