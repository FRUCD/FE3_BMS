130 bool NVM::load_hex_file(char *file)
131 {
132  if (sdReader.open(file)) // 0 == success, other equals some failure
133  return 0;
134 
135  reset_parse_hex();
136  if(processRecordsOnDemand(get_record) == 0)
137  {
138  return 0;
139  }
140  reset_parse_hex();
141 
142  // Wait for writing to finish because polling for ACK is
143  // failing in the smbus layer.
144  delay(100);
145  pmbus__->immediateOffGlobal();
146  delay(1000);
147  pmbus__->restoreFromNvmAll(addr, 3);
148  // Wait for the restore instead of polling for ACK.
149  delay(100);
150  pmbus__->sequenceOnGlobal();
151 
152  return 1;
153 }
